function resetDocument_(doc, body) {
  body.clear();
  while (body.getNumChildren() > 0) {
    body.removeChild(body.getChild(0));
  }
  const header = doc.getHeader();
  if (header) {
    header.clear();
  }
  const footer = doc.getFooter();
  if (footer) {
    footer.clear();
  }

  body
      .setPageWidth(PAGE.width)
      .setPageHeight(PAGE.height)
      .setMarginTop(PAGE.marginTop)
      .setMarginBottom(PAGE.marginBottom)
      .setMarginLeft(PAGE.marginLeft)
      .setMarginRight(PAGE.marginRight);
}

function applyInlineLinks_(text, fullText, links) {
  links.forEach(function(link) {
    const start = fullText.indexOf(link.label);
    if (start === -1) {
      return;
    }
    const end = start + link.label.length - 1;
    text
        .setLinkUrl(start, end, link.url)
        .setForegroundColor(start, end, STYLE.linkColor);
  });
}

function addHttps_(url) {
  return /^https?:\/\//.test(url) ? url : 'https://' + url;
}

function parseResumeSource_(source) {
  const sections = splitIntoSections_(source);
  return {
    header: parseHeader_(sections.__header__ || []),
    summary: parseParagraphs_(sections.SUMMARY || []),
    skills: parseSkillLines_(sections.SKILLS || []),
    employmentHistory: parseRoleEntries_(sections['EMPLOYMENT HISTORY'] || []),
    programsAndFellowships: parseRoleEntries_(sections['PROGRAMS & FELLOWSHIPS'] || []),
    projects: parseProjectEntries_(sections.PROJECTS || []),
    education: parseEducationEntries_(sections.EDUCATION || []),
  };
}

function splitIntoSections_(source) {
  const sectionNames = new Set([
    'SUMMARY',
    'SKILLS',
    'EMPLOYMENT HISTORY',
    'PROGRAMS & FELLOWSHIPS',
    'PROJECTS',
    'EDUCATION',
  ]);

  const sections = { __header__: [] };
  let current = '__header__';
  source.replace(/\r/g, '').split('\n').forEach(function(line) {
    const trimmed = line.trim();
    if (sectionNames.has(trimmed)) {
      current = trimmed;
      sections[current] = [];
      return;
    }
    sections[current].push(line);
  });
  return sections;
}

function parseHeader_(lines) {
  const compact = lines.map(function(line) {
    return line.trim();
  }).filter(Boolean);
  return {
    name: compact[0],
    location: compact[1],
    phone: compact[2],
    email: compact[3],
    links: compact[4].split(/\s+\|\s+/),
    headline: compact[5],
  };
}

function parseParagraphs_(lines) {
  return splitByBlankRuns_(lines, 1).map(joinWrappedLines_);
}

function parseSkillLines_(lines) {
  return lines.map(function(line) {
    return cleanLine_(line);
  }).filter(Boolean);
}

function parseRoleEntries_(lines) {
  const cleanLines = lines.map(cleanLine_).filter(Boolean);
  const entries = [];
  let current = null;

  for (let index = 0; index < cleanLines.length; index += 1) {
    if (isLikelyRoleHeaderAt_(cleanLines, index)) {
      if (current) {
        entries.push(finalizeRoleEntry_(current));
      }
      current = {
        title: cleanLines[index],
        org: cleanLines[index + 1],
        date: cleanLines[index + 2],
        bodyLines: [],
      };
      index += 2;
      continue;
    }

    if (!current) {
      continue;
    }

    current.bodyLines.push(cleanLines[index]);
  }

  if (current) {
    entries.push(finalizeRoleEntry_(current));
  }

  return entries;
}

function parseProjectEntries_(lines) {
  return splitByBlankRuns_(lines, 2).map(function(block) {
    const clean = block.map(cleanLine_).filter(Boolean);
    let bulletStart = clean.findIndex(function(line) {
      return /^•\s*/.test(line);
    });
    if (bulletStart === -1) {
      bulletStart = clean.length;
    }
    return {
      title: clean[0],
      description: joinWrappedLines_(clean.slice(1, bulletStart)),
      bullets: parseBullets_(clean.slice(bulletStart)),
    };
  });
}

function parseEducationEntries_(lines) {
  return splitByBlankRuns_(lines, 2).map(function(block) {
    const clean = block.map(cleanLine_).filter(Boolean);
    return {
      title: clean[0],
      org: clean[1],
      date: clean[2],
    };
  });
}

function splitByBlankRuns_(lines, blankRunForSplit) {
  const blocks = [];
  let current = [];
  let blankRun = 0;

  lines.forEach(function(line) {
    if (!line.trim()) {
      blankRun += 1;
      if (blankRun >= blankRunForSplit && current.length) {
        blocks.push(current);
        current = [];
      }
      return;
    }

    blankRun = 0;
    current.push(line);
  });

  if (current.length) {
    blocks.push(current);
  }

  return blocks;
}

function parseBullets_(lines) {
  const bullets = [];
  let current = [];

  lines.forEach(function(line) {
    const clean = cleanLine_(line);
    if (!clean) {
      return;
    }
    if (/^•\s*/.test(clean)) {
      if (current.length) {
        bullets.push(joinWrappedLines_(current));
      }
      current = [clean.replace(/^•\s*/, '')];
      return;
    }
    current.push(clean);
  });

  if (current.length) {
    bullets.push(joinWrappedLines_(current));
  }

  return bullets;
}

function joinWrappedLines_(lines) {
  return lines.map(function(line) {
    return cleanLine_(line);
  }).filter(Boolean).join(' ').replace(/\s+/g, ' ').trim();
}

function cleanLine_(line) {
  return String(line || '').replace(/\s+/g, ' ').trim();
}

function isLikelyRoleHeaderAt_(lines, index) {
  const title = lines[index];
  const org = lines[index + 1];
  const date = lines[index + 2];

  if (!title || !org || !date) {
    return false;
  }

  if (/^•\s*/.test(title) || /^•\s*/.test(org) || /^•\s*/.test(date)) {
    return false;
  }

  return looksLikeDateLine_(date);
}

function looksLikeDateLine_(line) {
  const normalized = normalizeForMatch_(line);
  if (!normalized) {
    return false;
  }

  if (/\bcurrent\b/.test(normalized)) {
    return true;
  }

  const monthPattern = /\b(jan|january|feb|february|mar|march|apr|april|may|jun|june|jul|july|aug|august|sep|sept|september|oct|october|nov|november|dec|december)\b/;
  return /\b(19|20)\d{2}\b/.test(normalized) && (monthPattern.test(normalized) || /\bto\b/.test(normalized) || /\bthrough\b/.test(normalized) || /\bcurrent\b/.test(normalized) || /\b\d{4}\b/.test(normalized));
}

function finalizeRoleEntry_(entry) {
  return {
    title: entry.title,
    org: entry.org,
    date: entry.date,
    bullets: parseBullets_(entry.bodyLines || []),
  };
}

function joinSentences_(parts) {
  return parts.filter(Boolean).join(' ').replace(/\s+/g, ' ').trim();
}

function compactProject_(project) {
  const featureText = project.bullets.map(function(bullet) {
    return bullet.replace(/\.$/, '');
  }).join(', ');
  return project.description.replace(/:\s*$/, ' ') + featureText + '.';
}

function logParsedResumeEntries_(resume) {
  logEntryGroup_('employmentHistory', resume.employmentHistory);
  logEntryGroup_('programsAndFellowships', resume.programsAndFellowships);
}

function logEntryGroup_(label, entries) {
  Logger.log('Parsed ' + label + ' entries: ' + entries.length);
  entries.forEach(function(entry, index) {
    Logger.log(
        label + '[' + index + '] title="' + safeDebugValue_(entry.title) +
        '" | org="' + safeDebugValue_(entry.org) +
        '" | date="' + safeDebugValue_(entry.date) +
        '" | bullets=' + ((entry.bullets || []).length));
  });
}

function safeDebugValue_(value) {
  return String(value || '').replace(/\s+/g, ' ').trim();
}

function findEntryByMarkers_(entries, label, markerGroups) {
  for (let entryIndex = 0; entryIndex < entries.length; entryIndex += 1) {
    const entry = entries[entryIndex];
    const haystack = normalizeForMatch_([entry.title, entry.org, entry.date].join(' '));
    if (matchesAnyMarkerGroup_(haystack, markerGroups)) {
      return entry;
    }
  }

  throw new Error(
      'Missing expected resume entry for "' + label + '". Tried marker groups: ' +
      formatMarkerGroups_(markerGroups) + '. Parsed entries: ' + describeEntries_(entries));
}

function findProjectByMarkers_(projects, label, markerGroups) {
  for (let index = 0; index < projects.length; index += 1) {
    const project = projects[index];
    const haystack = normalizeForMatch_([project.title, project.description].join(' '));
    if (matchesAnyMarkerGroup_(haystack, markerGroups)) {
      return project;
    }
  }

  throw new Error(
      'Missing expected project for "' + label + '". Tried marker groups: ' +
      formatMarkerGroups_(markerGroups) + '. Parsed projects: ' + describeProjects_(projects));
}

function matchesAnyMarkerGroup_(haystack, markerGroups) {
  return markerGroups.some(function(group) {
    return group.every(function(marker) {
      return haystack.indexOf(normalizeForMatch_(marker)) !== -1;
    });
  });
}

function normalizeForMatch_(text) {
  return String(text || '')
      .toLowerCase()
      .replace(/[—–-]/g, ' ')
      .replace(/[^a-z0-9]+/g, ' ')
      .replace(/\s+/g, ' ')
      .trim();
}

function formatMarkerGroups_(markerGroups) {
  return markerGroups.map(function(group) {
    return '[' + group.join(', ') + ']';
  }).join(' OR ');
}

function describeEntries_(entries) {
  return entries.map(function(entry, index) {
    return (
      '#' + index +
      ' title="' + safeDebugValue_(entry.title) +
      '" org="' + safeDebugValue_(entry.org) +
      '" date="' + safeDebugValue_(entry.date) + '"'
    );
  }).join(' ; ');
}

function describeProjects_(projects) {
  return projects.map(function(project, index) {
    return '#' + index + ' title="' + safeDebugValue_(project.title) + '"';
  }).join(' ; ');
}
